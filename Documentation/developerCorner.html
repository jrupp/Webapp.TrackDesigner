JSON Track Designer Layout (.jtdl) file format

A jtdl file as an utf-8 encoded json file, containing the following sections:


{
	/**
	 * file creation date
	 * @var integer date
	 */
	"date":1411918809691,
	
	/**
	 * file format
	 */
	"format":{
		/** @var integer version */
		"version":1,
		
		/** @var string jtdl file ending
		"type":"jtdl",
		
		/** @var string description */
		"description":"Json Track Designer Layout"
	},
	
	/**
	 * Information about the creation application
	 */
	"exportApplication":{
		/** @var string name */
		"name":"Webapp.TrackDesigner",
		
		/** @var string author */
		"author":"moonline",
		
		/** @var string url */
		"url":"https://github.com/moonline/Webapp.TrackDesigner"
	},
	
	/**
	 * Document license, only CC BY-NC-SA 3.0 is supported by Trackdesigner! */
	 * @var string license
	 */
	"license":"CC BY-NC-SA 3.0",
	
	/**
	 * Content section, containing shapes
	 * version 1 only supports shapes in section 'main'
	 */
	"content":{
		"layouts":{
			"main":{
				/**
				 * Layout
				 *
				 * @var string class
				 * @var number width
				 * @var number height
				 * @var Shape[] shapes
				 */
				"class":"Classes/Domain/Model/Layout",
				"width":2461.7150335387073,
				"height":1430.4195279629726,
				"shapes":[
					{
						/**
						 * Shape
						 *
						 * @var string class
						 * @var string type (id of ShapeType)
						 * @var string variant (id of VariantType)
						 * @var Point center
						 */
						"class":"Classes/Domain/Model/Shape",
						"type":"railRoadCrossing",
						"variant":"Grass",
						"center":{
							/**
							 * Point
							 *
							 * @var string class
							 * @var number x position
							 * @var number y position
							 * @var number angle -> turn (1=360Â°)
							 */
							"class":"Classes/Domain/Model/Point",
							"x":674.2,
							"y":260.14230313381245,
							"angle":0.5
						}
					},{
						"class":"Classes/Domain/Model/Shape",
						"type":"roadTCrossing",
						"variant":"Grass",
						"center":{
							"class":"Classes/Domain/Model/Point",
							"x":674.2,
							"y":515.3423031338125,
							"angle":0.5
						}
					}
					/* ... */
				]
			}
		}
	}
}
